66      protected abstract List getArgsList();
67      protected abstract String getCommandArgs();
68      private List getCommandLineArgs() throws MojoExecutionException
69      {
70      	List commandArguments = new ArrayList();
71      	
72  	    if ( getCommandArgs() != null )
73  	    {
74  	        String[] args = parseCommandlineArgs(getCommandArgs());
75  	        for ( int i = 0; i < args.length; i++ )
76  	        {
77  	             commandArguments.add( args[i] );
78  	        }
79  	    }
80  	    else if ( getArgsList() != null )
81  	    {
82  	        for ( int i = 0; i < getArgsList().size(); i++ )
83  	        {
84  	            Object argument = getArgsList().get( i );
85  	            String arg;
86  	            if ( argument == null )
87  	            {
88  	                throw new MojoExecutionException( "Misconfigured argument, value is null. "
89  	                    + "Set the argument to an empty value if this is the required behaviour." );
90  	            }
91  	            else
92  	            {
93  	                arg = argument.toString();
94  	                commandArguments.add( arg );
95  	            }
96  	        }
97  	    }
98  	    return commandArguments;
99      }

130     protected Map getEnvs()
131     {
132 	    Map enviro = new HashMap();
133 	    try
134 	    {
135 	        Properties systemEnvVars = CommandLineUtils.getSystemEnvVars();
136 	        enviro.putAll( systemEnvVars );
137 	    }
138 	    catch ( IOException x )
139 	    {
140 	        getLog().error( "Could not assign default system enviroment variables.", x );
141 	    }
142 	
143 	    if ( getMoreEnvironmentVariables() != null )
144 	    {
145 	        Iterator iter = getMoreEnvironmentVariables().keySet().iterator();
146 	        while ( iter.hasNext() )
147 	        {
148 	            String key = (String) iter.next();
149 	            String value = (String) getMoreEnvironmentVariables().get( key );
150 	            enviro.put( key, value );
151 	        }
152 	    }
153 	    return enviro;
154     }

181     CommandLine getExecutablePath( Map enviro, File dir )
182     {
183         File execFile = new File( getExecutable() );
184         String exec = null;
185         if ( execFile.exists() )
186         {
187             getLog().debug( "Toolchains are ignored, 'executable' parameter is set to " + getExecutable() );
188             exec = execFile.getAbsolutePath();
189         }
190         else
191         {
192             if ( OS.isFamilyWindows() )
193             {
194                 String ex = getExecutable().indexOf( "." ) < 0 ? getExecutable() + ".bat" : getExecutable();
195                 File f = new File( dir, ex );
196                 if ( f.exists() )
197                 {
198                     exec = ex;
199                 }
200                 else
201                 {
202                     // now try to figure the path from PATH, PATHEXT env vars
203                     // if bat file, wrap in cmd /c
204                     String path = (String) enviro.get( "PATH" );
205                     if ( path != null )
206                     {
207                         String[] elems = StringUtils.split( path, File.pathSeparator );
208                         for ( int i = 0; i < elems.length; i++ )
209                         {
210                             f = new File( new File( elems[i] ), ex );
211                             if ( f.exists() )
212                             {
213                                 exec = ex;
214                                 break;
215                             }
216                         }
217                     }
218                 }
219             }
220         }
221 
222         if ( exec == null )
223         {
224             exec = getExecutable();
225         }
226 
227         CommandLine toRet;
228         if ( OS.isFamilyWindows() && exec.toLowerCase( Locale.getDefault() ).endsWith( ".bat" ) )
229         {
230             toRet = new CommandLine( "cmd" );
231             toRet.addArgument( "/c" );
232             toRet.addArgument( exec );
233         }
234         else
235         {
236             toRet = new CommandLine( exec );
237         }
238 
239         return toRet;
240     }

252     protected boolean isResultCodeAFailure( int result )
253     {
254         if ( getSuccesCode() == null || getSuccesCode().size() == 0 ) 
255         {
256             return result != 0;
257         }
258         for ( Iterator it = getSuccesCode().iterator(); it.hasNext(); )
259         {
260             int code = Integer.parseInt( (String) it.next() );
261             if ( code == result ) 
262             {
263                 return false;
264             }
265         }
266         return true;
267     }

290     public void execute() throws MojoExecutionException
291     {
292         if ( isSkip() )
293         {
294             getLog().info( "skipping execute as per configuraion" );
295             return;
296         }
297 
298         if ( basedir == null )
299         {
300             throw new IllegalStateException( "basedir is null. Should not be possible." );
301         }
302 
303         List commandArguments = getCommandLineArgs();
304         
305         Map enviro = getEnvs();
306         
307         EnsureExistWorkingDirectory();
308 
309         CommandLine commandLine = getExecutablePath( enviro, getWorkingDir() );
310 
311         Executor exec = new DefaultExecutor();
312        
313         commandLine.addArguments( (String[])commandArguments.toArray(new String[commandArguments.size()]), false );
314 
315         exec.setWorkingDirectory( getWorkingDir() );
316 
317         try
318         {
319             getLog().info( "Executing command line: " + commandLine );
320             
321             preExecute(exec, commandLine, enviro);
322 
323             int resultCode = executeCommandLine( exec, commandLine, enviro, getOutputStreamOut(), getOutputStreamErr(), getInputStream());
324    
325             if ( isResultCodeAFailure( resultCode ) )
326             {
327                 throw new MojoExecutionException( "Result of " + commandLine + " execution is: '" + resultCode + "'." );
328             }
329             
330             postExecute(resultCode);
331         }
332         catch ( ExecuteException e )
333         {
334             throw new MojoExecutionException( "Command execution failed.", e );
335         }
336         catch ( IOException e )
337         {
338             throw new MojoExecutionException( "Command execution failed.", e );
339         }
