18  package org.codehaus.mojo;
19  
20  import java.io.File;
21  import java.io.FileOutputStream;
22  import java.io.IOException;
23  import java.io.InputStream;
24  import java.io.OutputStream;
25  import java.util.Arrays;
26  import java.util.Map;
27  
28  import org.apache.commons.exec.CommandLine;
29  import org.apache.commons.exec.DefaultExecutor;
30  import org.apache.commons.exec.ExecuteException;
31  import org.apache.commons.exec.Executor;
32  
33  import org.apache.maven.plugin.MojoExecutionException;
34  
35  /* Use FileSet and the FileManager provided in this project*/
36  import org.apache.maven.model.FileSet;
37  
38  /**
39   * Goal which gcovr execution.
40   *
41   * @author Franck Bonin 
42   * @goal coverage
43   * @phase test
44   * 
45   */
46  public class CoverageMojo extends LaunchMojo
47  {  
48      /**
49       * The Report OutputFile Location.
50       * 
51       * @parameter expression="${coverage.reportsfilePath}" default-value="gcovr-reports"
52       * @since 0.0.4
53       */
54      private File reportsfileDir;
55      
56      /**
57       * The Report OutputFile name identifier.
58       * 
59       * @parameter expression="${gcovr.reportIdentifier}" default-value=""
60       * @since 0.0.4
61       */
62      private String reportIdentifier;
63      
64      private String getReportFileName()
65      {
66          return "gcovr-result-" + reportIdentifier + ".xml";
67      }
68      
69      /**
70       * Arguments to clean preexisting gcda report under workingDir
71       * 
72       * @parameter expression="${gcovr.preclean}" default-value=true
73       */
74      private boolean preclean;
75      
76      protected void preExecute(Executor exec, CommandLine commandLine, Map enviro) throws MojoExecutionException
77      {
78          if ( preclean )
79          {
80              FileSet afileSet = new FileSet();
81              afileSet.setDirectory( getWorkingDir().getAbsolutePath() );
82              
83              getLog().debug( "Search for **/*.gcda from " + afileSet.getDirectory() );
84              afileSet.setIncludes( Arrays.asList( new String[]{"**/*.gcda"} ) );
85              //afileSet.setExcludes( Arrays.asList(excludes) );
86              
87              FileSetManager aFileSetManager = new FileSetManager();
88              String[] found = aFileSetManager.getIncludedFiles( afileSet );
89  
90              for (int i = 0; i < found.length; i++)
91              {
92                  File target = new File( getWorkingDir() + "/" + found[i]);
93                  getLog().debug( "Found file " + target.getAbsolutePath() );
94                  if (target.exists() )
95                  {
96                      try
97                      {
98                          if ( target.delete() )
99                          {
100                             getLog().debug( "Succesfully delete " + target.getAbsolutePath() );
101                         }
102                         else
103                         {
104                             getLog().warn( "Failed to delete " + target.getAbsolutePath() );
105                         }
106                     }
107                     catch ( SecurityException e )
108                     {
109                         getLog().warn( "SecurityException, unable to delete " + target.getAbsolutePath() );
110                     } 
111                 }
112                 else
113                 {
114                     getLog().debug( "But file " + target.getAbsolutePath() + " not exist" );
115                 }
116             }
117         }
118     }
119     
120     /**
121      * Arguments for the gcovr program. Shall be -x -d
122      * ex: -x -d to produce Xml reports and clean gcda execution reports after reading
123      * 
124      * @parameter expression="${gcovr.args}" default-value="-x -d"
125      */
126     private String gcovrArgs;
127     
128     protected void postExecute(int resultCode) throws MojoExecutionException
129     {
130         String OutputReportName = new String();
131         if ( reportsfileDir.isAbsolute() )
132         {
133             OutputReportName = reportsfileDir.getAbsolutePath() + "/" + getReportFileName();
134         }
135         else
136         {
137             OutputReportName = basedir.getAbsolutePath() + "/" + reportsfileDir.getPath() + "/" + getReportFileName();
138         }
139         getLog().info( "Coverage report location " + OutputReportName );
140          
141         OutputStream outStream = System.out;
142         File file = new File( OutputReportName );
143         try
144         {
145             new File( file.getParent() ).mkdirs();
146             file.createNewFile();
147             outStream = new FileOutputStream( file );
148         }
149         catch ( IOException e )
150         {
151             getLog().error( "Coverage report redirected to stdout since " + OutputReportName + " can't be opened" );
152         }
153         
154         InputStream pyScript = getClass().getResourceAsStream( "/gcovr.py" );
155         
156         CommandLine commandLine = new CommandLine( "python" );
157         Executor exec = new DefaultExecutor();
158         String[] args = parseCommandlineArgs( "-" );
159         commandLine.addArguments( args, false );
160         args = parseCommandlineArgs( gcovrArgs );
161         commandLine.addArguments( args, false );
162         exec.setWorkingDirectory( getWorkingDir() );
163         try
164         {
165             getLog().info( "Executing command line: " + commandLine );
166 
167             int res = executeCommandLine( exec, commandLine, getEnvs(), outStream/*getOutputStreamOut()*/, getOutputStreamErr(), pyScript/*getInputStream()*/ );
168             // this is a hugly workaround against a random bugs from hudson cobertura plugin.
169             // hudson cobertura plugin randomly truncat coverage reports file to a 1024 size multiple
170             // while it copy reports from slave to master node
171             for ( int j = 0 ; j < 200 ; j++ )
172             {
173                 for ( int i = 0 ; i < 80 ; i++ )
174                 {
175                     outStream.write( ' ' );
176                 }
177                 outStream.write( '\n' );
178             }
179             outStream.flush();
180             
181             if ( isResultCodeAFailure( res ) )
182             {
183                 throw new MojoExecutionException( "Result of command line execution is: '" + res + "'." );
184             }
185         }
186         catch ( ExecuteException e )
187         {
188             throw new MojoExecutionException( "Command execution failed.", e );
189         }
190         catch ( IOException e )
191         {
192             throw new MojoExecutionException( "Command execution failed.", e );
193         }
194     }
