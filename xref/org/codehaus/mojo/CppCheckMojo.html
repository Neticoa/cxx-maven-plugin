103     private String commandArgs;

105     protected String getCommandArgs()
106     {
107         String params = commandArgs + " ";
108         HashSet<String> excudedSet = new HashSet<String>();
109         
110         Iterator it = includeDirs.iterator();
111         while( it.hasNext() )
112         {
113             FileSet afileSet = new FileSet();
114             
115             String dir = it.next().toString();
116             params += "-I\"" + dir + "\" ";
117             
118             if ( StringUtils.isNotEmpty( excludes ) )
119             {
120                 afileSet.setDirectory( new File( dir ).getAbsolutePath() );
121                 afileSet.setUseDefaultExcludes( false ); // $FB pour éviter d'avoir TROP de fichier excludes (inutiles) dans la boucle for ci-après
122                 afileSet.setExcludes( Arrays.asList( excludes.split( "," ) ) );
123                 getLog().debug( "cppcheck excludes are :" + Arrays.toString( afileSet.getExcludes().toArray() ) );
124                 
125                 FileSetManager aFileSetManager = new FileSetManager();
126                 String[] found = aFileSetManager.getExcludedFiles( afileSet );
127                 excudedSet.addAll( new HashSet<String>( Arrays.asList( found ) ));
128             }
129         }
130         it = sourceDirs.iterator();
131         while( it.hasNext() )
132         {
133             FileSet afileSet = new FileSet();
134             
135             String dir = it.next().toString();
136             params += "-I\"" + dir + "\" ";
137             
138             if ( StringUtils.isNotEmpty( excludes ) )
139             {
140                 afileSet.setDirectory( new File( dir ).getAbsolutePath() );
141                 afileSet.setUseDefaultExcludes( false ); // $FB pour éviter d'avoir TROP de fichiers exclude (inutile) dans la boucle for ci-après
142                 afileSet.setExcludes( Arrays.asList( excludes.split( "," ) ) );
143                 getLog().debug( "cppcheck excludes are :" + Arrays.toString( afileSet.getExcludes().toArray() ) );
144                 
145                 FileSetManager aFileSetManager = new FileSetManager();
146                 String[] found = aFileSetManager.getExcludedFiles( afileSet);
147                 excudedSet.addAll( new HashSet<String>( Arrays.asList( found) ));
148             }
149         }
150         for ( Iterator<String> iter = excudedSet.iterator(); iter.hasNext(); )
151         {
152             String s = iter.next();
153             //cppcheck only check *.cpp, *.cxx, *.cc, *.c++, *.c, *.tpp, and *.txx files
154             // so remove unneeded exclusions
155             if (s.matches( "(.+\\.cpp)|(.+\\.cxx)|(.+\\.cc)|(.+\\.c\\+\\+)|(.+\\.c)|(.+\\.tpp)|(.+\\.txx)" ) )
156             {
157                 params += "-i\"" + s + "\" ";
158             }
159         }
160         
161         it = sourceDirs.iterator();
162         while( it.hasNext() )
163         {
164             params += "\"" + it.next() + "\" ";
165         }
166         
167         return params;
168     }

170     //override
171     protected OutputStream getOutputStreamErr()
172     {
173         String OutputReportName = new String();
174         if ( reportsfileDir.isAbsolute() )
175         {
176             OutputReportName = reportsfileDir.getAbsolutePath() + "/" + getReportFileName();
177         }
178         else
179         {
180             OutputReportName = basedir.getAbsolutePath() + "/" + reportsfileDir.getPath() + "/" + getReportFileName();
181         }
182         
183         getLog().info( "Cppcheck report location " + OutputReportName );
184          
185         OutputStream output = System.err;
186         File file = new File( OutputReportName );
187         try
188         {
189             new File( file.getParent() ).mkdirs();
190             file.createNewFile();
191             output = new FileOutputStream( file );
192         }
193         catch ( IOException e )
194         {
195             getLog().error( "Cppcheck report redirected to stderr since " + OutputReportName + " can't be opened" );
196         }
197 
198         return output;
199     }
