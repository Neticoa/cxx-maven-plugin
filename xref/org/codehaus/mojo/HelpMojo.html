562     /**
563      * <p>Repeat a String <code>n</code> times to form a new string.</p>
564      *
565      * @param str String to repeat
566      * @param repeat number of times to repeat str
567      * @return String with repeated String
568      * @throws NegativeArraySizeException if <code>repeat < 0</code>
569      * @throws NullPointerException if str is <code>null</code>
570      */
571     private static String repeat( String str, int repeat )
572     {
573         StringBuffer buffer = new StringBuffer( repeat * str.length() );
574 
575         for ( int i = 0; i < repeat; i++ )
576         {
577             buffer.append( str );
578         }
579 
580         return buffer.toString();
581     }

583     /** 
584      * Append a description to the buffer by respecting the indentSize and lineLength parameters.
585      * <b>Note</b>: The last character is always a new line.
586      * 
587      * @param sb The buffer to append the description, not <code>null</code>.
588      * @param description The description, not <code>null</code>.
589      * @param indent The base indentation level of each line, must not be negative.
590      */
591     private void append( StringBuffer sb, String description, int indent )
592     {
593         for ( Iterator it = toLines( description, indent, indentSize, lineLength ).iterator(); it.hasNext(); )
594         {
595             sb.append( it.next().toString() ).append( '\n' );
596         }
597     }

599     /** 
600      * Splits the specified text into lines of convenient display length.
601      * 
602      * @param text The text to split into lines, must not be <code>null</code>.
603      * @param indent The base indentation level of each line, must not be negative.
604      * @param indentSize The size of each indentation, must not be negative.
605      * @param lineLength The length of the line, must not be negative.
606      * @return The sequence of display lines, never <code>null</code>.
607      * @throws NegativeArraySizeException if <code>indent < 0</code>
608      */
609     private static List toLines( String text, int indent, int indentSize, int lineLength )
610     {
611         List lines = new ArrayList();
612 
613         String ind = repeat( "\t", indent );
614         String[] plainLines = text.split( "(\r\n)|(\r)|(\n)" );
615         for ( int i = 0; i < plainLines.length; i++ )
616         {
617             toLines( lines, ind + plainLines[i], indentSize, lineLength );
618         }
619 
620         return lines;
621     }

623     /** 
624      * Adds the specified line to the output sequence, performing line wrapping if necessary.
625      * 
626      * @param lines The sequence of display lines, must not be <code>null</code>.
627      * @param line The line to add, must not be <code>null</code>.
628      * @param indentSize The size of each indentation, must not be negative.
629      * @param lineLength The length of the line, must not be negative.
630      */
631     private static void toLines( List lines, String line, int indentSize, int lineLength )
632     {
633         int lineIndent = getIndentLevel( line );
634         StringBuffer buf = new StringBuffer( 256 );
635         String[] tokens = line.split( " +" );
636         for ( int i = 0; i < tokens.length; i++ )
637         {
638             String token = tokens[i];
639             if ( i > 0 )
640             {
641                 if ( buf.length() + token.length() >= lineLength )
642                 {
643                     lines.add( buf.toString() );
644                     buf.setLength( 0 );
645                     buf.append( repeat( " ", lineIndent * indentSize ) );
646                 }
647                 else
648                 {
649                     buf.append( ' ' );
650                 }
651             }
652             for ( int j = 0; j < token.length(); j++ )
653             {
654                 char c = token.charAt( j );
655                 if ( c == '\t' )
656                 {
657                     buf.append( repeat( " ", indentSize - buf.length() % indentSize ) );
658                 }
659                 else if ( c == '\u00A0' )
660                 {
661                     buf.append( ' ' );
662                 }
663                 else
664                 {
665                     buf.append( c );
666                 }
667             }
668         }
669         lines.add( buf.toString() );
670     }

672     /** 
673      * Gets the indentation level of the specified line.
674      * 
675      * @param line The line whose indentation level should be retrieved, must not be <code>null</code>.
676      * @return The indentation level of the line.
677      */
678     private static int getIndentLevel( String line )
679     {
680         int level = 0;
681         for ( int i = 0; i < line.length() && line.charAt( i ) == '\t'; i++ )
682         {
683             level++;
684         }
685         for ( int i = level + 1; i <= level + 4 && i < line.length(); i++ )
686         {
687             if ( line.charAt( i ) == '\t' )
688             {
689                 level++;
690                 break;
691             }
692         }
693         return level;
694     }
