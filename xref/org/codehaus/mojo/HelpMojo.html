737      * 
738      * @param text The text to split into lines, must not be <code>null</code>.
739      * @param indent The base indentation level of each line, must not be negative.
740      * @param indentSize The size of each indentation, must not be negative.
741      * @param lineLength The length of the line, must not be negative.
742      * @return The sequence of display lines, never <code>null</code>.
743      * @throws NegativeArraySizeException if <code>indent < 0</code>
744      */
745     private static List toLines( String text, int indent, int indentSize, int lineLength )
746     {
747         List lines = new ArrayList();
748 
749         String ind = repeat( "\t", indent );
750         String[] plainLines = text.split( "(\r\n)|(\r)|(\n)" );
751         for ( int i = 0; i < plainLines.length; i++ )

759     /** 
760      * Adds the specified line to the output sequence, performing line wrapping if necessary.
761      * 
762      * @param lines The sequence of display lines, must not be <code>null</code>.
763      * @param line The line to add, must not be <code>null</code>.
764      * @param indentSize The size of each indentation, must not be negative.
765      * @param lineLength The length of the line, must not be negative.
766      */
767     private static void toLines( List lines, String line, int indentSize, int lineLength )
768     {
769         int lineIndent = getIndentLevel( line );
770         StringBuffer buf = new StringBuffer( 256 );
771         String[] tokens = line.split( " +" );
772         for ( int i = 0; i < tokens.length; i++ )
773         {
774             String token = tokens[i];
775             if ( i > 0 )
776             {
777                 if ( buf.length() + token.length() >= lineLength )
778                 {
779                     lines.add( buf.toString() );
780                     buf.setLength( 0 );
781                     buf.append( repeat( " ", lineIndent * indentSize ) );
782                 }
783                 else
784                 {
785                     buf.append( ' ' );
786                 }
787             }
788             for ( int j = 0; j < token.length(); j++ )
789             {
790                 char c = token.charAt( j );
791                 if ( c == '\t' )
792                 {
793                     buf.append( repeat( " ", indentSize - buf.length() % indentSize ) );
794                 }
795                 else if ( c == '\u00A0' )
796                 {
797                     buf.append( ' ' );
798                 }
799                 else
800                 {
801                     buf.append( c );
802                 }
803             }
804         }
805         lines.add( buf.toString() );
806     }

808     /** 
809      * Gets the indentation level of the specified line.
810      * 
811      * @param line The line whose indentation level should be retrieved, must not be <code>null</code>.
812      * @return The indentation level of the line.
813      */
814     private static int getIndentLevel( String line )
815     {
816         int level = 0;
817         for ( int i = 0; i < line.length() && line.charAt( i ) == '\t'; i++ )
818         {
819             level++;
820         }
821         for ( int i = level + 1; i <= level + 4 && i < line.length(); i++ )
822         {
823             if ( line.charAt( i ) == '\t' )
824             {
825                 level++;
826                 break;
827             }
828         }
829         return level;
830     }
