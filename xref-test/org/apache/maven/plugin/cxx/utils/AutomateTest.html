35  {
36      // @formatter:off
37      /**
38       * We implement this funny automate :
39       * </code>
40       *     good/good_vibration()
41       *     +---------------------+       +---------+
42       * +---+----+                |   +---+----+    | good/good_vibration()
43       * |        |                +-->|        |    | 
44       * | INIT   |                    | FINAL  |<---+ 
45       * |choose()|                +-->|        |
46       * +---+----+                |   +--------+          
47       *     | bad/bad_vibration() | 
48       *     +---------------------+
49       * 
50       * Init state 'choose' something...
51       * </code>
52       * @author fbonin
53       * 
54       */
55      // @formatter:on
56      static class Sample extends Automate {
57  
58          /**
59           * define automate labels
60           */
61          enum MainLabel 
62          {
63              bad,
64              good,
65              other;
66  
67              /**
68               * TODO 
69               * 
70               * @param i
71               * @return enum
72               */
73              public static MainLabel fromOrdinal( final int i )
74              {
75                  if ( ( i < 0 ) || ( i >= MainLabel.values().length ) )
76                  {
77                      throw new IndexOutOfBoundsException( "Invalid ordinal" );
78                  }
79                  return MainLabel.values()[i];
80              }
81          };
82  
83          /**
84           * define automate states
85           */
86          enum MainState {
87              FINAL,
88              INIT;
89  
90              /**
91               * 
92               * @param i
93               * @return enum
94               */
95              public static MainState fromOrdinal(final int i)
96              {
97                  if ((i < 0) || (i >= MainState.values().length))
98                  {
99                      throw new IndexOutOfBoundsException("Invalid ordinal");
100                 }
101                 return MainState.values()[i];
102             }
103         };
104 
105         /**
106          * Automate definition
107          */
108         static {
109             Action good_vibration = new Action()
110             {
111                 public void action(final ActionExecutor executor, final Object aiParam)
112                     throws AutomateException
113                 {
114 
115                     ((Sample) executor).good_vibration(aiParam);
116                 }
117 
118                 public String getName()
119                 {
120                     return "good_vibration";
121                 }
122             };
123             Action bad_vibration = new Action()
124             {
125                 public void action(final ActionExecutor executor, final Object aiParam)
126                     throws AutomateException
127                 {
128                     ((Sample) executor).bad_vibration(aiParam);
129                 }
130 
131                 public String getName()
132                 {
133                     return "bad_vibration";
134                 }
135             };
136             Action choose = new Action()
137             {
138                 public void action(final ActionExecutor executor, final Object aiParam)
139                                         throws AutomateException
140                 {
141                     ((Sample) executor).choose(aiParam);
142                 }
143 
144                 public String getName()
145                 {
146                     return "choose";
147                 }
148             };
149             try
150             {
151                 Automate.initializeAutomate(MainState.values().length, MainLabel.values().length);
152                 Automate.setTransition(MainState.INIT.ordinal(), MainLabel.good.ordinal(),
153                                         MainState.FINAL.ordinal(), good_vibration);
154                 Automate.setTransition(MainState.INIT.ordinal(), MainLabel.bad.ordinal(),
155                                         MainState.FINAL.ordinal(), bad_vibration);
156                 Automate.setTransition(MainState.FINAL.ordinal(), MainLabel.good.ordinal(),
157                                         MainState.FINAL.ordinal(), good_vibration);
158                 Automate.setState(MainState.INIT.ordinal(), choose);
159             }
160             catch (Exception e)
161             {
162                 e.printStackTrace();
163             }
164         }
165 
166         /**
167          * Default constructor enter in initial state executor is
168          * Sample itself
169          * 
170          * @throws AutomateException
171          */
172         public Sample() throws AutomateException
173         {
174             super(MainState.INIT.ordinal(), "Parameter 1");
175         }
176 
177         /**
178          * Constructor to restore from any state executor is Sample
179          * itself
180          * 
181          * @param aiCurrentState
182          * @throws AutomateException
183          */
184         public Sample(final MainState aiCurrentState) throws AutomateException
185         {
186             super(aiCurrentState.ordinal(), "Parameter 2");
187         }
188 
189         /**
190          * TODO
191          *

194         public void bad_vibration(final Object s)
195         {
196             System.out.println("Bad Vibration transition has access to parameter : " + s);
197         }
198 
199         /**
200          * TODO
201          * 
202          * @param s
203          * @throws AutomateException
204          */
205         public void choose(final Object s) throws AutomateException
206         {
207             // this is a state action that choose its own label to
208             // follow :
209             MainLabel l = MainLabel.bad;
210             System.out.println("choose state action choose " + l + " label");
211             // don't forget to let automate know of our choice
212             event(l.ordinal(), s);
213         }

218             return "sample";
219         }

226         {
227             System.out.println("Good Vibration transition has access to parameter : " + s);
228         }
229 
230         @Override
231         protected boolean isInFinalState()
232         {
233             return getCurrentState() == MainState.FINAL.ordinal();
234         }
235

237         protected void notifyCurrentStateChanged()
238         {
239             System.out.println("State changed to : " + MainState.fromOrdinal(getCurrentState()));
240         }
241     };

259     /**
260      * @throws Exception if any
261      */
262     public void testAutomateBinaire() throws Exception
263     {
264         // choose_your_vibration() will be called since we step
265         // in initial state with its auto-choose action
266         Sample m = new Sample();
267         assertEquals(Sample.MainState.fromOrdinal(m.getCurrentState()),
268                             Sample.MainState.FINAL);
269         // 'good' event is allowed in init and final state
270         m.event(Sample.MainLabel.good.ordinal(), "parameter 3");
271         assertEquals(Sample.MainState.fromOrdinal(m.getCurrentState()),
272                             Sample.MainState.FINAL);
273     }
